<script>
    document.getElementById('themeToggle').addEventListener('click', () => {
        document.body.classList.toggle('light');
    });

    document.getElementById('enterBtn').addEventListener('click', () => {
        document.getElementById('landingPage').style.display = 'none';
        document.getElementById('mainContent').style.display = 'block';
    });
    document.getElementById('contactBtn').addEventListener('click', () => {
        window.open('https://wa.me/33652128731', '_blank');
    });

    document.getElementById('extractBtn').addEventListener('click', async () => {
        const url = document.getElementById('urlInput').value.trim();
        const resultDiv = document.getElementById('result');
        const resultContent = document.getElementById('resultContent');

        if (!url) {
            alert('Entrez un lien valide');
            return;
        }

        resultDiv.style.display = 'block';
        resultContent.innerHTML = '<div class="loader"></div> Extraction en cours...';

        try {
            // Utilisation de l'API Piped (gratuite, sans clé)
            const response = await fetch(`https://pipedapi.kavin.rocks/streams?url=${encodeURIComponent(url)}`);
            const data = await response.json();
            
            // Chercher la meilleure qualité vidéo
            const video = data.videoStreams?.find(s => s.quality === '720p') || 
                         data.videoStreams?.find(s => s.quality === '480p') || 
                         data.videoStreams?.[0];
            
            if (video && video.url) {
                resultContent.innerHTML = `
                    ✅ Extraction réussie !<br><br>
                    📹 Titre : ${data.title || 'Vidéo'}<br>
                    🎬 Qualité : ${video.quality || 'HD'}<br><br>
                    <a href="${video.url}" target="_blank" style="color:#10b981; font-size:1.2rem;">📥 Cliquer ici pour télécharger</a><br><br>
                    <button id="directDownloadBtn" style="margin-top:10px; background:#10b981;">⬇️ Télécharger directement</button>
                    <p style="margin-top:15px; font-size:0.8rem;">💡 Conseil : faites un clic droit → "Enregistrer sous" si nécessaire</p>
                `;
                document.getElementById('directDownloadBtn')?.addEventListener('click', () => {
                    window.open(video.url, '_blank');
                });
            } else {
                throw new Error('Aucune vidéo trouvée');
            }
        } catch (err) {
            console.error(err);
            resultContent.innerHTML = `
                ❌ Extraction impossible pour ce lien.<br><br>
                📌 <strong>Liens qui fonctionnent :</strong><br>
                • YouTube : https://www.youtube.com/watch?v=...<br>
                • TikTok : https://www.tiktok.com/@user/video/...<br>
                • Instagram : https://www.instagram.com/reel/...<br><br>
                🔗 Alternative : <a href="https://en.savefrom.net/" target="_blank" style="color:#3b82f6;">SaveFrom.net</a>
            `;
        }
    });
</script>
